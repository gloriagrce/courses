##   `2011` = col_double()
## )
staff_longer <- staff %>% 
  pivot_longer(
    cols = -faculty_type,
    names_to = "year",
    values_to = "percentage"
    )
  
  
staff_longer %>% 
  ggplot((mapping = aes(x = percentage, y = faculty_type, fill =  year))) +
           geom_col(position = "dodge")+
  scale_fill_viridis_d()+
  labs(title = "Instructional staff employment trends, 1975-2011",
       x = "% of faculty",
       caption = "Source: AAUP",
       y = NULL,
       fill = NULL)
